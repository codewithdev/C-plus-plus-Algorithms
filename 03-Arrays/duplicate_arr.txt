vector<int> duplicates(int arr[], int n); // creating function duplicates which returns a vector

int main() {
    int t; // t is number of testcases
    cin >> t;
    while (t--) {
        int n; // n is length of array
        cin >> n;
        int a[n];
        for (int i = 0; i < n; i++) 
		cin >> a[i];
        vector<int> ans = duplicates(a, n); // calling function duplicates
        for (int i : ans) cout << i << ' ';
        cout << endl;
    }
    return 0;
}

vector<int> duplicates(int arr[], int n) {
vector<int> v;
    sort(arr,arr+n); // we sort the array
    int i=0;
    int j;
    while(i<n-1){
        j=i+1;
        if(arr[j]==arr[i])
            v.push_back(arr[i]);
        while(arr[i]==arr[j])
            j++;
        i=j;
    }
    
    if(v.empty())  // if no element is repeating we return -1 as output
        v.push_back(-1);
    return v;
    }
